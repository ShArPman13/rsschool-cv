function qSort(array) {
  let length = array.length - 1
  if (array.length <= 1) {
    return array
  }
  let pivot = array[length];
  let low = [];
  let more = [];
   
  for (let i = 0; i < length; i += 1) {
    array[i] < pivot ? low.push(array[i]) : more.push(array[i]);
  }
  return [...qSort(low), pivot, ...qSort(more)];
}
